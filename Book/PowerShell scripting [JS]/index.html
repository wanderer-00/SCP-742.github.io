Скрипты на JS для PowerShell

этот скрипт эмулирует нажатия клавишь клавиатуры

// КЛАВИШИ, НАЖАТИЯ КОТОРЫХ МУЛИРУЕМ
var str = "человек по натуре склонен к саморазрушеню";

// ПОДКЛЮЧЕНИЕ ИСПОЛНЯЙМОЙ ЧАСТИ POWER SHELL
var WshShell;
WshShell = WScript.CreateObject("WScript.Shell");

// ЖДАТЬ [ms]
WScript.Sleep (5000);

array = str.split('');
var e1 = array.length;
//WshShell.SendKeys(array[1]);

for (i=0; i < e1; i++){
//console.log(array[i]);
WshShell.SendKeys(array[i]);
WScript.Sleep (4);
};

// ОКОШКО С ТЕКСТОМ И КНОПКОЙ 'ОК'
WScript.Echo ("конец");
