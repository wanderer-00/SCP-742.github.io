ОБНУЛЕНИЕ стилей

*
/* ОБНУЛЕНИЕ СТИЛЕЙ */
*,
*:after,
*:before
{
    margin: 0;
    padding: 0;
    box-sizing: border-box;
}
button
/* ОБНУЛЕНИЕ СТИЛЕЙ */
button {
    padding: 0;
    border: none;
    font: inherit;
    color: inherit;
    background-color: transparent;
}
a
/* ОБНУЛЕНИЕ СТИЛЕЙ */
a {text-decoration: none}
old new
input, select, button, a:
/* Универсальное решение, которое убирает подсветку во всех браузерах и не только на input, но и на других элементах, в том числе select, button, a: */
:active,
:hover,
:focus {
    outline: 0;
    outline-offset: 0;
    box-shadow: 0;/* УДАЛЕНИЕ ТЕНИ ДЛЯ ЭТИХ ПСЕВДОКЛАССОВ */
}

настройка scrollbar

/* SCROLLBAR */
::-webkit-scrollbar {/* 1 - скроллбар */}
::-webkit-scrollbar-butto {/* 2 - кнопка */}
::-webkit-scrollbar-track {/* 3 - трек */}
::-webkit-scrollbar-track-piece {/* 4 - видимая часть трека */}
::-webkit-scrollbar-thumb {/* 5 - ползунок */}
::-webkit-scrollbar-corner {/* 6 - уголок */}
::-webkit-resizer {/* 7 - изменение размеров */}
пример
/* SCROLL BAR */
::-webkit-scrollbar {width: 8px}
::-webkit-scrollbar-track {background: #18191d}
::-webkit-scrollbar-thumb {background: #3b4146}
::-webkit-scrollbar-thumb:hover {background: #474e54}
